/* @todo pass jquery.rating object here as the only param. create jquery.rating.submit() that is publicly available for call and call it inside onSubmit instead of passing callback */
var checkVote = function (rating) {
    var popup = jQuery('#captchaPopup'),
        self = this,
        showCaptcha = function () {
            SOLAR.captcha.reload();
            popup.sdialog({
                'autoOpen':      true,
                'title':         'Pass the test to vote',
                'disableSubmit': true,
                'modal':         true,
                'buttons': {
                    'ok': {
                        'id':    'button-submit',
                        'text':  'OK',
                        'click': function (e) {
                            if (e) {
                                e.preventDefault();
                            }

                            rating.options.captcha = $("#rating-captcha").val();
                            rating.mask();
                            rating.sendToServer();
                            popup.sdialog('close');
                        }
                    },
                    'cancel': {
                        'id':    'button-cancel',
                        'text':  'Cancel',
                        'click': function (e) {
                            popup.sdialog('close');
                        }
                    }
                }
            });
        };

    jQuery.ajax(SOLAR.ajax.getSettings({
        'type':    'POST',
        'url':     'https://www.solarmovie.ph/link/pre-vote/',
        'data':    { 'id': self.id },
        'successAny': function (response, context) {
            if (response.data.pass){
                rating.mask();
                rating.sendToServer();
            } else {
                showCaptcha();
            }
        }
    }));
};

    'quality':        'video',
    'resultDivT':     '.js-result .js-dynamic',
    'onSubmit' :      checkVote
